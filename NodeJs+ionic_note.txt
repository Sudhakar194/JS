> keytool -genkey -v -keystore my-testrelease-key.jks -keyalg RSA -keysize 2048 -validity 10000 -alias my-alias

> jarsigner -verbose -sigalg SHA1withRSA -digestalg SHA1 -keystore my-testrelease-key.jks <path - D:\Sudhakar\WorkSpaces - NodeJS\WorkSpace\myappsidemenu\platforms\android\app\build\outputs\apk\debug>/app-testrelease-unsigned.apk my-alias
