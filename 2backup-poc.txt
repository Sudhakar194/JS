            function myMove() {

              var elem = document.getElementById("myAnimation");   
              var pos = 0;
              var id = setInterval(frame, 10);
              var elemPos = false;
              function frame() {
                if (elemPos) {
                   clearInterval(id);
                    if (!elemPos){
                      var id = setInterval(frame, 10);    
                    }
                  pos--;
                  elem.style.left    = pos - 'px';
                  elem.style.top  = pos - 'px';        
                  if (pos == 0){
                       elemPos = false; 
                   }  
                } else {
                  pos++; 
                  elem.style.top  = pos + 'px'; 
                  elem.style.left = pos + 'px'; 
                  if (pos == 250){
                       elemPos = true; 
                   }
                }
              }
            }
